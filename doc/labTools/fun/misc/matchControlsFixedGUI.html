 matchControlsFixedGUI  Fixed GUI to choose demographic matching factors
 Compatible with MATLAB R2024a.

 GUI fields shown: Sex, Age (via DOB), Height, Weight, Years formal

 Computes weighted distance across numeric variables and returns the top-2

  - Tolerances for height/weight are also converted into canonical unit.
  - Adjusted layout: results area given more vertical space; left-panel
    Weight / Tolerance / Data-units columns narrowed for readability.

0444 % helper function: normalize categorical values for reliable comparison
0445 % - special handling for Sex: maps common short codes to canonical labels.
0446 % - for other categorical variables we use trimmed lowercase strings.
0447 % -----------------------------
0448     function out = normalizeCategorical(vals,varKey)
0449         % normalize input values (vals can be string array, char, numeric, cell)
0450         % varKey: name of the variable (e.g., 'sex','race') to allow special-casing
0451         s = string(vals);              % convert to string array (handles mixed types)
0452         s = lower(strtrim(s));         % lowercase and trim whitespace
0453         % replace missing with empty string
0454         s(ismissing(s)) = "";
0455         out = s;
0456         % special mapping for sex-like fields
0457         if contains(lower(varKey),'sex')
0458             femaleTokens = ["f","female","fem","woman","w"];
0459             maleTokens   = ["m","male","man"];
0460             otherTokens  = ["other","o","nonbinary","non-binary","nb","genderqueer","trans","nonbinary"];
0461             for ii = 1:numel(out)
0462                 v = out(ii);
0463                 if v=="" , continue; end
0464                 cleaned = regexprep(char(v),'[^a-z0-9]',''); % strip punctuation/spaces
0465                 cleaned = string(cleaned);
0466                 if any(cleaned == femaleTokens)
0467                     out(ii) = "female";
0468                 elseif any(cleaned == maleTokens)
0469                     out(ii) = "male";
0470                 elseif any(cleaned == otherTokens)
0471                     out(ii) = "other";
0472                 else
0473                     out(ii) = cleaned; % fallback lowercase cleaned token
0474                 end
0475             end
0476         else
0477             % generic cleaning for other categorical values
0478             for ii = 1:numel(out)
0479                 v = out(ii);
0480                 if v=="" , continue; end
0481                 cleaned = regexprep(char(v),'^\s+|\s+$','');    % trim ends
0482                 cleaned = regexprep(cleaned,'\s+',' ');        % collapse interior whitespace
0483                 out(ii) = lower(string(cleaned));
0484             end
0485         end
0486     end

0490 % - Note: convertUnits(varName, fromUnit, toUnit, vals)
0491 % - Supports height units: meters, centimeters, inches
0492 % - Supports weight units: kilograms, pounds
0493 % -----------------------------
0494     function out = convertUnits(varName, fromUnit, toUnit, vals)
0495         out = vals;
0496         if isempty(fromUnit) || isempty(toUnit) || strcmpi(fromUnit,toUnit), return; end
0497         lname = lower(varName);
0498         if contains(lname,'height')
0499             % normalize: convert 'fromUnit' values into meters, then to 'toUnit'
0500             switch lower(fromUnit)
0501                 case 'meters'
0502                     meters = vals;
0503                 case 'centimeters'
0504                     meters = vals/100;
0505                 case 'inches'
0506                     meters = vals*0.0254;
0507                 case {'feet', 'ft'}
0508                     meters = vals*0.3048;
0509                 case {'feet and inches', 'ftin', 'ft/in'}
0510                     % Expect vals as Nx2: [feet, inches] or a single value (total inches)
0511                     if ismatrix(vals) && size(vals,2) == 2
0512                         meters = vals(:,1)*0.3048 + vals(:,2)*0.0254;
0513                     else
0514                         meters = vals*0.0254; % fallback: treat as total inches
0515                     end
0516                 otherwise
0517                     meters = vals; % assume already meters if unknown
0518             end
0519             switch lower(toUnit)
0520                 case 'meters'
0521                     out = meters;
0522                 case 'centimeters'
0523                     out = meters*100;
0524                 case 'inches'
0525                     out = meters/0.0254; % meters -> inches
0526                 case {'feet', 'ft'}
0527                     out = meters/0.3048;
0528                 case {'feet and inches', 'ftin', 'ft/in'}
0529                     % Return Nx2: [feet, inches]
0530                     feet = floor(meters/0.3048);
0531                     inches = (meters - feet*0.3048)/0.0254;
0532                     out = [feet, inches];
0533                 otherwise
0534                     out = meters;
0535             end
0536         elseif contains(lname,'weight')
0537             % convert 'fromUnit' values into kilograms, then to 'toUnit'
0538             switch lower(fromUnit)
0539                 case 'kilograms', kg = vals;
0540                 case 'pounds', kg = vals*0.45359237;
0541                 otherwise, kg = vals; % assume kilograms if unknown
0542             end
0543             switch lower(toUnit)
0544                 case 'kilograms', out = kg;
0545                 case 'pounds', out = kg / 0.45359237;
0546                 otherwise, out = kg;
0547             end
0548         end
