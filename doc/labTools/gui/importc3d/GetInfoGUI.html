GetInfoGUI

PURPOSE

GetInfoGUI GUI for collecting experimental session information.

SYNOPSIS

function varargout = GetInfoGUI(varargin)

DESCRIPTION 

 GetInfoGUI  GUI for collecting experimental session information.

   Launches a graphical user interface to gather all information regarding
 a single experiment conducted in the Sensorimotor Learning Laboratory.
 Fields include participant demographics, experiment metadata, data file
 locations, trial and condition assignments, and EMG channel labels. Refer
 to in-GUI help text (hover over any field) for field-specific guidance.

   Outputs:
     info - Struct containing all session information entered by the user,
            or empty ([]) if the GUI was closed without saving.

   Toolbox Dependencies:
     None

   See also: errorProofInfo, importc3d/ExpDetails

1456 % --- Executes on button press in saveExpButton.
1457 function saveExpButton_Callback(hObject, eventdata, handles)
1458 % saveExpButton_Callback  Saves the current condition configuration as
1459 %   a new experiment description file in the ExpDetails directory.
1460 %
1461 %   Inputs:
1462 %     hObject   - handle to saveExpButton (see GCBO)
1463 %     eventdata - reserved for future MATLAB versions
1464 %     handles   - struct with handles and user data (see GUIDATA)
1465 
1466 % Build the expDes structure from current GUI field values
1467 nConds = 0;
1468 for ii = 1:handles.lines
1469     % Condition numbers
1470     condNum = get(handles.(['condition' num2str(ii)]), 'String');
1471     if ~isempty(condNum)
1472         expDes.(['condition' num2str(ii)]) = condNum;
1473         nConds = nConds + 1;
1474     end
1475     % Condition names
1476     condName = get(handles.(['condName' num2str(ii)]), 'String');
1477     if ~isempty(condName)
1478         expDes.(['condName' num2str(ii)]) = condName;
1479     end
1480     % Condition descriptions
1481     condDesc = get(handles.(['description' num2str(ii)]), 'String');
1482     if ~isempty(condDesc)
1483         expDes.(['description' num2str(ii)]) = condDesc;
1484     end
1485     % Trial numbers for each condition
1486     trialNum = get(handles.(['trialnum' num2str(ii)]), 'String');
1487     if ~isempty(trialNum)
1488         expDes.(['trialnum' num2str(ii)]) = trialNum;
1489     end
1490     % Trial types
1491     type = get(handles.(['type' num2str(ii)]), 'String');
1492     if ~isempty(type)
1493         expDes.(['type' num2str(ii)]) = type;
1494     end
1495 end
1496 expDes.numofconds = nConds;
1497 
1498 answer = inputdlg( ...
1499     'Enter name of new experiment description: ', ...
1500     'Experiment Description Name');
1501 if ~isempty(answer)
1502     answer = char(answer);
1503     expDes.group = answer;
1504     % Remove non-alphanumeric characters from the filename
1505     answer = answer(ismember(answer, ['A':'Z' 'a':'z' '0':'9']));
1506     detailsPath = which('GetInfoGUI');
1507     detailsPath = strrep(detailsPath, 'GetInfoGUI.m', 'ExpDetails');
1508     if isfile(fullfile(detailsPath, [answer '.mat']))
1509         choice = questdlg( ...
1510             'File name already exists. Overwrite?', ...
1511             'File Name Warning', 'Yes', 'No', 'No');
1512         if strcmp(choice, 'No')
1513             h = msgbox('Experiment description was not saved.', '');
1514             waitfor(h);
1515             return;
1516         end
1517     end
1518     save(fullfile(detailsPath, answer), 'expDes');
1519     description_edit_CreateFcn(handles.description_edit, ...
1520         eventdata, handles);
1521     newContents = get(handles.description_edit, 'String');
1522     ind = find(ismember(newContents, answer));
1523     set(handles.description_edit, 'Value', ind);
1524 end
