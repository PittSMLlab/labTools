getStanceFromForces

PURPOSE

%% If Fz is sampled >2000Hz, downsample to somewhere in the [1000-2000)Hz

SYNOPSIS

function [stance] = getStanceFromForces(Fz, threshold, fsample)

DESCRIPTION 

  %% If Fz is sampled >2000Hz, downsample to somewhere in the [1000-2000)Hz
  %range (for computational cost reduction, no other reason).
  
  M=1;
  if fsample>2000
      M=floor(fsample/1000);
      Fz=Fz(1:M:end);
      fsample=fsample/M;
  end
